q4] Implement a function that checks whether a given number is prime or not.


// function for prime

fn is_prime(num: u64) -> bool {
    if num < 2 {
        return false;
    }
    for i in 2..=num/2 {
        if num % i == 0 {
            return false;
        }
    }
    true
}




// main function

fn main() {
    let num = 17;
    if is_prime(num) {
        println!("{} is prime", num);
    } else {
        println!("{} is not prime", num);
    }
}
